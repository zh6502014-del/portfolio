⚙️ 設定 Supabase 連線

在開始使用後台之前，請先完成 Supabase 設定。這是免費的雲端資料庫，用來儲存您的作品集內容與圖片。

1
前往 supabase.com 免費註冊並建立新專案
2
在 Project Settings → API 找到 Project URL 和 anon public key
3
在 SQL Editor 執行下方的建表語法
4
在 Storage 建立名為 portfolio-images 的 bucket，設定為 Public
📋 SQL 建表語法（點擊展開） 
-- 個人資料
CREATE TABLE profile (
  id SERIAL PRIMARY KEY,
  name TEXT,
  tagline TEXT,
  description TEXT,
  email TEXT,
  phone TEXT,
  note TEXT,
  status TEXT DEFAULT '目前在職',
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- 技能
CREATE TABLE skills (
  id SERIAL PRIMARY KEY,
  title TEXT NOT NULL,
  icon TEXT DEFAULT '🎨',
  tags TEXT[] DEFAULT '{}',
  sort_order INT DEFAULT 0
);

-- 工具
CREATE TABLE tools (
  id SERIAL PRIMARY KEY,
  name TEXT NOT NULL,
  description TEXT,
  sort_order INT DEFAULT 0
);

-- 專案
CREATE TABLE projects (
  id SERIAL PRIMARY KEY,
  title TEXT NOT NULL,
  time_range TEXT,
  tags TEXT[] DEFAULT '{}',
  description TEXT,
  role TEXT,
  members TEXT,
  goals TEXT[] DEFAULT '{}',
  challenges TEXT[] DEFAULT '{}',
  methods TEXT[] DEFAULT '{}',
  work_items TEXT[] DEFAULT '{}',
  products TEXT[] DEFAULT '{}',
  insight TEXT,
  images TEXT[] DEFAULT '{}',
  sort_order INT DEFAULT 0,
  published BOOLEAN DEFAULT TRUE,
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- 隨筆
CREATE TABLE notes (
  id SERIAL PRIMARY KEY,
  title TEXT NOT NULL,
  note_date DATE,
  note_type TEXT,
  content TEXT,
  sort_order INT DEFAULT 0,
  published BOOLEAN DEFAULT TRUE
);

-- RLS (開放匿名讀取，只有登入者可寫)
ALTER TABLE profile ENABLE ROW LEVEL SECURITY;
ALTER TABLE skills ENABLE ROW LEVEL SECURITY;
ALTER TABLE tools ENABLE ROW LEVEL SECURITY;
ALTER TABLE projects ENABLE ROW LEVEL SECURITY;
ALTER TABLE notes ENABLE ROW LEVEL SECURITY;

CREATE POLICY "public read" ON profile FOR SELECT USING (true);
CREATE POLICY "auth write" ON profile FOR ALL USING (auth.role() = 'authenticated');
CREATE POLICY "public read" ON skills FOR SELECT USING (true);
CREATE POLICY "auth write" ON skills FOR ALL USING (auth.role() = 'authenticated');
CREATE POLICY "public read" ON tools FOR SELECT USING (true);
CREATE POLICY "auth write" ON tools FOR ALL USING (auth.role() = 'authenticated');
CREATE POLICY "public read" ON projects FOR SELECT USING (true);
CREATE POLICY "auth write" ON projects FOR ALL USING (auth.role() = 'authenticated');
CREATE POLICY "public read" ON notes FOR SELECT USING (true);
CREATE POLICY "auth write" ON notes FOR ALL USING (auth.role() = 'authenticated');
